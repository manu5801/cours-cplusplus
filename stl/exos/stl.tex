\subsection*{\underline{Exercice 1}: Tableau dynamique - vector}

\begin{itemize}
	\item [1.] Allouez 5 entiers au vecteur vect
    \item [2.] Ajoutez les entiers $6,7$ et $8$ à la fin de vect
    \item [3.] Supprimez le dernier élément 
    \item [4.] Affichez le nombre d'éléments qu'il contient
    \item [5.] Parcourir à l'aide d'une boucle les valeurs de vect, affichez les.
\end{itemize}


\vspace{.5cm}

\subsection*{\underline{Exercice 2}: List }

\begin{itemize}
	\item [1.] Ajoutez à la fin de la liste les éléments suivants : $5, 6, 1, 10, 7, 8, 4, 5$
    \item [2.] Affichez la taille de la liste
    \item [3.] Utilisez un itérateur pour parcourir la liste et affichez les éléments
    \item [4.] Affichez le premier et le dernier élément
    \item [5.] Parcourez avec un itérateur en inverse la liste
\end{itemize}

\subsection*{\underline{Exercice 3}: Map }
\begin{itemize}
	\item [1.] Remplissez la table associative (map) : mois et nombre de jours du mois
    \item [2.] Affichez la taille de la map
    \item [3.] Utilisez un itérateur pour parcourir la map et affichez les mois avec le nombre de jours correspondant 
    \item [4.] Affichez le nombre de jours du mois de Janvier
    \item [5.] Affichez les jours du mois de Janvier
\end{itemize}

\vspace{.5cm}


\subsection*{\underline{Exercice 4}: FILE ou QUEUE : FIFO }
\begin{itemize}
	\item [1.] Ajoutez les nombres $1,4,2$ dans la file
    \item [2.] Affichez la taille de la file
    \item [3.] En faisant défiler les éléments de la file, affichez l'élément le plus ancien et supprimez le.
    \item[4.] Affichez de nouveau la taille de la file
\end{itemize}

\vspace{.5cm}


\subsection*{\underline{Exercice 5}: PILE ou STACK : LIFO }
\begin{itemize}
	\item [1.] Ajoutez les nombres $1,4,2$ dans la pile
    \item [2.] Affichez la taille de la pile
    \item [3.] En faisant défiler les éléments de la pile, affichez l'élément en haut de la pile et supprimez le.
    \item[4.] Affichez de nouveau la taille de la pile
\end{itemize}

\vspace{.5cm}


\subsection*{\underline{Exercice 6}: Tri d'un tableau - Sort }
\begin{itemize}
	\item [1.] Ajoutez les nombres $23, 5, -10, 0, 0, 321, 1, 2, 99, 30$ dans le tableau
    \item [2.] Affichez les éléments du tableau
    \item [3.] Triez le tableau à l'aide de la fonction sort $http://www.cplusplus.com/reference/algorithm/sort/$
    \item[4.] Affichez de nouveau les éléments du tableau
\end{itemize}

\vspace{.5cm}

\subsection*{\underline{Exercice 7}: Tri décroissant d'un vector -Sort }
\begin{itemize}
	\item [1.] Ajoutez les nombres $23, 5, -10, 0, 0, 321, 1, 2, 99, 30$ dans le vector
    \item [2.] Affichez les éléments du vector
    \item [3.] Faites un tri décroissant des éléments du vector à l'aide de la fonction sort $http://www.cplusplus.com/reference/algorithm/sort/$
    \item[4.] Affichez de nouveau les éléments du vector
\end{itemize}

\subsection*{\underline{Exercice 8}: Désallocation mémoire }

\end{itemize}
